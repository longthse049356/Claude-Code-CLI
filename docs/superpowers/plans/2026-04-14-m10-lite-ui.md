# M10 Lite UI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a React dashboard UI (4 panels: Channel, Chat, Agent, Log) that talks to the M3 Bun server via REST + WebSocket.

**Architecture:** Server-side adds 2 new routes (`GET /channels`, `GET /channels/:id/agents`), a logger wrapper that broadcasts `{ type: "log" }` WS events, and static file serving from `packages/ui/dist/`. Frontend is React + Vite scaffolded in `packages/ui/`, using Zustand for realtime WS state and TanStack Query for REST. In dev, Vite proxies API + WS to Bun server at port 3456. In production, `bun run build:ui` outputs static files served by Bun at port 3456.

**Tech Stack:** Bun, TypeScript, React 18, Vite, Shadcn/ui, Tailwind CSS, Zustand, TanStack Query v5

```typescript
async function serveStatic(pathname: string): Promise<Response> {
  const distDir = join(import.meta.dir, "../packages/ui/dist");
  // Sanitize: prevent path traversal attacks
  const safePath = pathname.replace(/^(\.\.(\/|\\|$))+/, "").replace(/^\//, "") || "index.html";
  const filePath = join(distDir, safePath);
  const file = Bun.file(filePath);
  if (await file.exists()) return new Response(file);
  // SPA fallback — unknown routes serve index.html (React Router handles them)
  return new Response(Bun.file(join(distDir, "index.html")));
}
```

Update the `fetch` handler to call `serveStatic` when the router returns 404:

```typescript
  fetch(req, server) {
    const url = new URL(req.url);
    const pathname = url.pathname;

    if (pathname === "/browser/ws") return new Response(null, { status: 404 });

    if (pathname === "/ws") {
      log(`[SERVER] WS upgrade request`);
      const upgraded = server.upgrade(req, { data: undefined });
      if (!upgraded) {
        log(`[SERVER] WS upgrade FAILED`);
        return new Response("WS upgrade failed", { status: 400 });
      }
      return;
    }

    log(`\n[SERVER] ${req.method} ${pathname}`);
    const response = handleRequest(req);
    return response.then((res) => {
      if (res.status !== 404) return res;
      return serveStatic(pathname);
    });
  },
```

```typescript
// packages/ui/src/stores/useThemeStore.ts
import { create } from "zustand";

type Theme = "dark" | "light";

function applyTheme(theme: Theme): void {
  const root = document.documentElement;
  root.classList.remove("dark", "light");
  root.classList.add(theme);
}

interface ThemeStore {
  theme: Theme;
  toggle: () => void;
}

const saved = (localStorage.getItem("theme") as Theme | null) ?? "dark";

export const useThemeStore = create<ThemeStore>((set, get) => ({
  theme: saved,
  toggle: () => {
    const next: Theme = get().theme === "dark" ? "light" : "dark";
    localStorage.setItem("theme", next);
    applyTheme(next);
    set({ theme: next });
  },
}));

// Apply saved theme immediately on store init
applyTheme(saved);
```

```typescript
// packages/ui/src/stores/useWsStore.ts
import { create } from "zustand";
import type { DbMessage } from "../types";

const MAX_LOGS = 500;

interface WsStore {
  connected: boolean;
  messages: Record<string, DbMessage[]>; // channelId → messages[]
  typingAgents: { agent_name: string; channel_id: string }[];
  logs: string[];
  // actions
  setConnected: (v: boolean) => void;
  seedMessages: (channelId: string, msgs: DbMessage[]) => void;
  addMessage: (msg: DbMessage) => void;
  setTyping: (data: { agent_name: string; channel_id: string }) => void;
  clearTyping: (channelId: string) => void;
  addLog: (entry: string) => void;
}

export const useWsStore = create<WsStore>((set) => ({
  connected: false,
  messages: {},
  typingAgents: [],
  logs: [],

  setConnected: (v) => set({ connected: v }),

  seedMessages: (channelId, msgs) =>
    set((s) => ({
      messages: { ...s.messages, [channelId]: msgs },
    })),

  addMessage: (msg) =>
    set((s) => {
      const existing = s.messages[msg.channel_id] ?? [];
      // Deduplicate by id — WS may arrive while TanStack refetch is in flight
      if (existing.some((m) => m.id === msg.id)) return s;
      return {
        messages: {
          ...s.messages,
          [msg.channel_id]: [...existing, msg],
        },
      };
    }),

  setTyping: (data) =>
    set((s) => ({
      typingAgents: [
        ...s.typingAgents.filter((t) => t.agent_name !== data.agent_name),
        data,
      ],
    })),

  clearTyping: (channelId) =>
    set((s) => ({
      typingAgents: s.typingAgents.filter((t) => t.channel_id !== channelId),
    })),

  addLog: (entry) =>
    set((s) => ({
      logs: [...s.logs.slice(-(MAX_LOGS - 1)), entry],
    })),
}));
```

```typescript
// packages/ui/src/hooks/useWebSocket.ts
import { useEffect, useRef } from "react";
import type { WsBroadcast } from "../types";
import { useWsStore } from "../stores/useWsStore";

export function useWebSocket(): void {
  const { setConnected, addMessage, setTyping, addLog } = useWsStore();
  const retryDelay = useRef(1000);

  useEffect(() => {
    let ws: WebSocket;
    let stopped = false;

    function connect(): void {
      const protocol = window.location.protocol === "https:" ? "wss:" : "ws:";
      ws = new WebSocket(`${protocol}//${window.location.host}/ws`);

      ws.onopen = () => {
        setConnected(true);
        retryDelay.current = 1000;
      };

      ws.onclose = () => {
        setConnected(false);
        if (!stopped) {
          setTimeout(() => {
            retryDelay.current = Math.min(retryDelay.current * 2, 30000);
            connect();
          }, retryDelay.current);
        }
      };

      ws.onmessage = (e: MessageEvent<string>) => {
        try {
          const event = JSON.parse(e.data) as WsBroadcast;
          if (event.type === "new_message") addMessage(event.data);
          else if (event.type === "typing") setTyping(event.data);
          else if (event.type === "log") addLog(event.data);
        } catch {
          // ignore malformed messages
        }
      };
    }

    connect();
    return () => {
      stopped = true;
      ws?.close();
    };
  }, []); // mount once
}
```

```typescript
import { useEffect } from "react";
import { useQuery } from "@tanstack/react-query";
import type { DbMessage } from "../types";
import { useWsStore } from "../stores/useWsStore";

async function fetchMessages(channelId: string): Promise<DbMessage[]> {
  const res = await fetch(`/channels/${channelId}/messages`);
  if (!res.ok) throw new Error("Failed to fetch messages");
  return res.json();
}

export function useMessages(channelId: string | null) {
  const seedMessages = useWsStore((s) => s.seedMessages);

  const query = useQuery({
    queryKey: ["messages", channelId],
    queryFn: () => fetchMessages(channelId!),
    enabled: channelId !== null,
  });

  // Seed WsStore so new WS messages merge with history
  useEffect(() => {
    if (query.data && channelId) {
      seedMessages(channelId, query.data);
    }
  }, [query.data, channelId, seedMessages]);

  return query;
}
```

```tsx
  const [sending, setSending] = useState(false);
  const bottomRef = useRef<HTMLDivElement>(null);

  const selectedChannelId = useAppStore((s) => s.selectedChannelId);
  const messages = useWsStore((s) =>
    selectedChannelId ? (s.messages[selectedChannelId] ?? []) : []
  );

      setText("");

      setSending(false);
    }
  }

  if (!selectedChannelId) {
    return (
      <div className="flex-1 flex items-center justify-center text-muted-foreground text-sm">

    );

    <div className="flex flex-col h-full">
      {/* Messages */}
      <ScrollArea className="flex-1 px-4 py-3">

          <div
            key={msg.id}
            className={`mb-3 ${msg.role === "user" ? "text-right" : "text-left"}`}

            </div>
          </div>
        ))}

        {typingAgents.map((t) => (
          <div key={t.agent_name} className="mb-3 text-left">
            <span className="inline-block px-3 py-2 rounded-lg text-sm bg-muted text-muted-foreground italic">
              {t.agent_name} is typing…
            </span>
          </div>

          onChange={(e) => setText(e.target.value)}
```
